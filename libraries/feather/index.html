Animations

 Just add the classes "animate", and add the name of the animation! 
 <!-- Link to Feather -->
<link rel="stylesheet" type="text/css" href="http://kingpixil.github.io/Feather/css/feather.css" />
<script src="http://kingpixil.github.io/Feather/js/feather.js"></script>
<!-- Make a row with everything in the center -->
<div class=" row text-center ">
  <div class="animate bounce infinite ">
    <h1>Feather</h1>
  </div>
  <div class="row text-center">
    <!-- Buttons => Add class btn + btn-outline, btn-3d, btn-round, or btn-gradient -->
    <button class="btn btn-3d ">Click Me</button>
  </div>
</div>
<!-- Make a script to do something when the button is clicked, using Feather.js -->
<script>
  $('button').on("click", function(event) {
  	alert("This is FEATHER.js");
    if ($("button ").hasClass("btn-3d")) {
    $("button").removeClass('btn-3d');
    $("button").addClass("btn-gradient");
    } else {
    $("button").removeClass("btn-gradient");
    $("button").addClass("btn-3d");
    };
  });
</script>
